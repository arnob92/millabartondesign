function doPost(e) {
  // Set CORS headers
  const response = ContentService.createTextOutput();
  response.setMimeType(ContentService.MimeType.JSON);
  
  try {
    // Parse the incoming data
    const jsonData = JSON.parse(e.postData.contents);
    const formType = jsonData.formType;
    const formData = jsonData.formData;
    Logger.log(formType);
    
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    
    if (formType === 'contact') {
      const contactSheet = ss.getSheetByName('Contact');
      const contactData = [
        new Date(),
        formData.name,
        formData.email,
        formData.subject,
        formData.message,
      ];
      contactSheet.appendRow(contactData);
      response.setContent(JSON.stringify({status: 'success'}));
    } 
    else if (formType === 'quickquote') {
      const quoteSheet = ss.getSheetByName('QuickQuote');
      const quoteData = [
        new Date(),
        formData.projectType?.join(', ') || '',
        formData.numberOfRooms?.join(', ') || '',
        formData.roomType?.join(', ') || '',
        formData.surfaceArea || 0,
        formData.designStyle?.join(', ') || '',
        formData.postalCode || '',
        formData.firstName || '',
        formData.email || '',
        formData.phone || ''
      ];
      quoteSheet.appendRow(quoteData);
      response.setContent(JSON.stringify({status: 'success'}));
    }
    else {
      throw new Error('Invalid form type');
    }
    
    return response;
    
  } catch (error) {
    response.setContent(JSON.stringify({
      status: 'error',
      message: error.message
    }));
    return response;
  }
}
